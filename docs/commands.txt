

# Debug Polkit
$>systemctl stop polkit
$>G_MESSAGES_DEBUG=all /usr/lib/polkit-1/polkitd

# Add network
$>nmcli connection add type ethernet con-name eth0 ifname eth0 ipv4.method manual ipv4.address 10.10.10.10 ipv4.gateway 10.10.10.1 ipv4.dns 10.10.10.1 ipv4.route-metric 100

# Control network
$>nmcli con up eth0
$>nmcli con down eth0
